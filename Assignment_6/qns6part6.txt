    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Temperature Converter</title>
    <script>
        function celsiusToFahrenheit() {
            let celsius = parseFloat(document.getElementById("celsius").value);

            if (isNaN(celsius)) {
                document.getElementById("result").innerHTML = "Please enter a valid temperature.";
                return;
            }

            let fahrenheit = (celsius * 9/5) + 32;
            document.getElementById("result").innerHTML = 
                `${celsius}°C is equal to ${fahrenheit.toFixed(2)}°F`;
        }
    </script>
</head>
<body>
    <h2>Convert Celsius to Fahrenheit</h2>

    <label for="celsius">Enter Temperature in Celsius:</label>
    <input type="number" id="celsius"><br><br>

    <button onclick="celsiusToFahrenheit()">Convert</button>

    <p id="result"></p>
</body>
